# Program to calculate the sum of numbers from 1 to 50 using a for loop

# Step 1: Initialize sum
total_sum = 0

# Step 2: Iterate over numbers from 1 to 50
for num in range(1, 51):
    total_sum += num   # Add each number to the sum

# Step 3: Display the final sum
print("The sum of numbers from 1 to 50 is:", total_sum)
